\section{Factorial}

% label a0000000003
\begin{definition}[Factorial]
  \label{def:factorial}
  \lean{fac}
  \leanok
  For any natural number $n$, we define the factorial of $n$, denoted by $n!$,
  recursively as:
  \begin{itemize}
    \item $0! = 1$
    \item $(n+1)! = (n+1) \cdot n!$
  \end{itemize}
\end{definition}

% label a0000000004
\begin{theorem}[Positivity of Factorials]
  \label{thm:fac_pos}
  \lean{fac_pos}
  \leanok
  For every natural number $n$, the factorial of $n$ is strictly positive:
  \[ n! > 0 \]
\end{theorem}

\begin{proof}
  \leanok
  \uses{fac}
  By induction on $n$.
  \begin{itemize}
    \item Base case: $0! = 1 > 0$.
    \item Inductive step: Assume $n! > 0$. Then $(n+1)! = (n+1) \cdot n!$.
      Since $n+1 > 0$ and $n! > 0$ (by inductive hypothesis), their product is positive.
  \end{itemize}
\end{proof}
